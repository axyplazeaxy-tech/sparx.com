// 🎮 Controller Support for Five Nights Custom let controllerConnected = false; let previousButtons = []; let activeController = null; window.addEventListener("gamepadconnected", function(e){ controllerConnected = true; activeController = e.gamepad.index; previousButtons = []; console.log( "Controller connected:", e.gamepad.id ); }); window.addEventListener("gamepaddisconnected", function(e){ controllerConnected = false; activeController = null; previousButtons = []; console.log( "Controller disconnected" ); }); function buttonPressed(pad, button){ return ( pad.buttons[button] && pad.buttons[button].pressed && !previousButtons[button] ); } function checkController(){ if(!controllerConnected){ requestAnimationFrame(checkController); return; } let pad = navigator.getGamepads()[activeController]; if(!pad){ requestAnimationFrame(checkController); return; } // A / Cross = Cameras if(buttonPressed(pad,0)){ openCamera(); } // B / Circle = Back if(buttonPressed(pad,1)){ closeCamera(); } // X / Square = Light if(buttonPressed(pad,2)){ toggleLight(); } // Y / Triangle = Door if(buttonPressed(pad,3)){ toggleDoor(); } previousButtons = pad.buttons.map( button => button.pressed ); requestAnimationFrame(checkController); } checkController();
